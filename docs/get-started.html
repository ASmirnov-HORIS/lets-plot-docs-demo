Quick Start

val rand = java.util.Random() val data = mapOf ( "rating" to List(200) { rand.nextGaussian() } + List(200) { rand.nextGaussian() * 1.5 + 1.5 }, "cond" to List(200) { "A" } + List(200) { "B" } ) letsPlot(data) + geomDensity(color = "dark_green", alpha = .3) { x = "rating"; fill = "cond" } + ggsize(700, 350)
